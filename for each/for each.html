// JavaScript's forEach() is a method that allows you to iterate over arrays // It calls a function ONCE for each element in the array (in order) // Syntax: array.forEach(callback(element, index, array)) const fruits = ['ðŸŽ', 'ðŸŒ', 'ðŸ‡', 'ðŸ“']; // Example 1: Basic forEach fruits.forEach(function(fruit, index) { // The callback function receives: // 1. fruit - the current element // 2. index - the index of the current element // 3. array - the whole array (not used here, but available) console.log(`Fruit at index ${index} is ${fruit}`); }); // Example 2: Using arrow functions (shorter syntax) fruits.forEach((fruit, index) => { console.log(`[Arrow] Index: ${index}, Fruit: ${fruit}`); }); // Example 3: Perform an action with forEach const numbers = [1, 2, 3, 4, 5]; let sum = 0; numbers.forEach(num => { sum += num; // Accumulate the sum }); console.log(`Sum of numbers: ${sum}`); // Sum: 15 // Example 4: forEach does NOT return a new array (unlike map or filter) const result = numbers.forEach(num => num * 2); console.log(`Result: ${result}`); // undefined // Example 5: You CANNOT break out of forEach with break or return (use for-loop or some if needed) numbers.forEach(num => { if (num === 3) { // This return only exits the current callback, not the whole loop console.log(`Found ${num}, but cannot break forEach`); return; } console.log(`Processing ${num}`); }); // Example 6: forEach on objects using Object.keys const person = { name: 'Oswald', age: 19, goal: 'Master JavaScript' }; Object.keys(person).forEach(key => { console.log(`${key}: ${person[key]}`); }); // Summary in comments: // - forEach is great for performing an action on every array element. // - Cannot break/return early. Use for-loops for that. // - It does NOT create a new array (unlike map, filter, reduce). // - Can pass element, index, and array to the callback.
